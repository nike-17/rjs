Начало работы с Russian JS

В этом руководстве вы узнаете, как начать использовать Russian JS Transpiler в своём проекте.

Установка​

Глобальная установка​

Установите RJS глобально для использования из командной строки:

npm install -g russian-js-transpiler

Установка как зависимости​

Или установите как dev-зависимость в ваш проект:

npm install --save-dev russian-js-transpiler

Базовое использование​

Командная строка​

Транспиляция файла​

rjs compile входной_файл.ржс -o выходной_файл.js

Наблюдение за изменениями​

rjs watch исходники/ -o сборка/

Прямое выполнение​

rjs run скрипт.ржс

Использование в JavaScript​

const { transpile } = require('russian-js-transpiler');

const rjsCode = 'переменная привет = "Привет, мир!";';
const jsCode = transpile(rjsCode);

console.log(jsCode);
// Выведет: var privet = "Привет, мир!";

Интеграция с Node.js​

Создайте файл app.ржс:

// app.ржс
функция приветствовать(имя) {
  вернуть `Привет, ${имя}!`;
}

консоль.лог(приветствовать("Мир"));

Установите rjs-node:

npm install -g rjs-node

Запустите файл:

rjs-node app.ржс

Интеграция с веб-пакетами​

Webpack​

Установите загрузчик:

npm install --save-dev rjs-loader

Добавьте в webpack.config.js:

module.exports = {
  // ...
  module: {
    rules: [
      {
        test: /\.ржс$/,
        use: 'rjs-loader',
        exclude: /node_modules/,
      },
    ],
  },
};

Rollup​

Установите плагин:

npm install --save-dev rollup-plugin-rjs

Добавьте в rollup.config.js:

import rjs from 'rollup-plugin-rjs';

export default {
  // ...
  plugins: [
    rjs()
  ]
};

Конфигурационный файл​

Создайте файл rjs.config.js в корне проекта:

module.exports = {
  // Входной каталог
  input: 'src',
  
  // Выходной каталог
  output: 'dist',
  
  // Расширения файлов для обработки
  extensions: ['.rjs', '.рус'],
  
  // Минификация выходного кода
  minify: true,
  
  // Генерация source maps
  sourceMaps: true,
  
  // Настройки транспиляции
  compilerOptions: {
    // Сохранять русские имена переменных
    preserveVariableNames: false,
    
    // Уровень логирования
    logLevel: 'info',
    
    // Настройки совместимости
    target: 'es2020',
  }
};

Примеры​

Пример 1: Простой калькулятор​

// calculator.ржс
функция калькулятор() {
  консоль.лог("Простой калькулятор");
  консоль.лог("Доступные операции: +, -, *, /");
  
  переменная а = число(подсказка("Введите первое число:"));
  переменная операция = подсказка("Введите операцию (+, -, *, /):");
  переменная б = число(подсказка("Введите второе число:"));
  
  переменная результат;
  
  переключатель (операция) {
    случай "+":
      результат = а + б;
      перерыв;
    случай "-":
      результат = а - б;
      перерыв;
    случай "*":
      результат = а * б;
      перерыв;
    случай "/":
      если (б === 0) {
        консоль.ошибка("Ошибка: деление на ноль!");
        вернуть;
      }
      результат = а / б;
      перерыв;
    по_умолчанию:
      консоль.ошибка("Неизвестная операция");
      вернуть;
  }
  
  консоль.лог(`Результат: ${а} ${операция} ${б} = ${результат}`);
}

калькулятор();

Пример 2: Работа с API​

// weather.ржс
асинхронная функция получитьПогоду(город) {
  попробовать {
    константа API_KEY = "ВАШ_API_КЛЮЧ";
    константа URL = `https://api.openweathermap.org/data/2.5/weather?q=${город}&appid=${API_KEY}&units=metric`;
    
    константа ответ = ждать fetch(URL);
    
    если (!ответ.ок) {
      выбросить новая Ошибка(`Ошибка HTTP: ${ответ.статус}`);
    }
    
    константа данные = ждать ответ.вJson();
    вернуть данные;
  } поймать (ошибка) {
    консоль.ошиб("Не удалось получить данные о погоде:", ошибка.сообщение);
    вернуть ноль;
  }
}

// Использование
(асинхронная () => {
  константа город = подсказка("Введите город:");
  константа погода = ждать получитьПогоду(город);
  
  если (погода) {
    консоль.лог(`\nПогода в ${город}:`);
    консоль.лог(`Температура: ${погода.основное.температура}°C`);
    консоль.лог(`Влажность: ${погода.основное.влажность}%`);
    консоль.лог(`Скорость ветра: ${погода.ветер.скорость} м/с`);
    консоль.лог(`Описание: ${погода.погода[0].описание}`);
  }
})();
