## Переменные и константы

```javascript
// Объявление константы
константа ПИ = 3.14159;

// Объявление переменной
переменная счётчик = 0;

const { имя, возраст } = получитьПользователя();
```

## Условные операторы

### Базовые условия

```javascript
если (возраст >= 18) {
  консоль.лог("Доступ разрешён");
} иначе если (возраст >= 16) {
  консоль.лог("Доступ с ограничениями");
} иначе {
  консоль.лог("Доступ запрещён");
}

// Тернарный оператор
сообщение = возраст >= 18 ? "Взрослый" : "Ребёнок";

// Оператор нулевого слияния
значение = переменная ?? "значение по умолчанию";
```

### Switch-case

```javascript
переключить (деньНедели) {
  случай 1:
    консоль.лог("Понедельник");
    прервать;
  случай 5:
    консоль.лог("Пятница");
    прервать;
  по_умолчанию:
    консоль.лог("Другой день");
}
```

## Циклы

### Цикл for

```javascript
// Классический for
для (пусть i = 0; i < 10; i++) {
  если (i === 5) продолжить;
  если (i === 8) прервать;
  консоль.лог(i);
}

// Итерация по объекту
для (пусть ключ в объект) {
  консоль.лог(`${ключ}: ${объект[ключ]}`);
}
```

### Цикл while

```javascript
// Цикл с предусловием
пока (счётчик < 10) {
  консоль.лог(счётчик);
  счёттор++;
}

// Цикл с постусловием
делать {
  счётчик--;
} пока (счёттор > 0);
```

### Итерация по коллекциям

```javascript
// for...of для массивов и итерируемых объектов
для (элемент из массива) {
  консоль.лог(элемент);
}

// for...in для перебора свойств объекта
для (свойство в объект) {
  консоль.лог(`${свойство}: ${объект[свойство]}`);
}
```

## Классы

```javascript
класс Животное {
  конструктор(имя) {
    this.имя = имя;
  }

  говори() {
    консоль.лог(`${this.имя} издает звук.`);
  }
}

класс Собака расширяет Животное {
  лает() {
    консоль.лог('Гав-гав!');
  }
}

const мойПес = новый Собака('Рекс');
мойПес.говори(); // Рекс издает звук.
мойПес.лает();    // Гав-гав!
```
