# Синтаксис

## Переменные

```javascript
пусть число = 42;
конст приветствие = "Привет, мир!";
```

## Условия

```javascript
если (возраст >= 18) {
  консоль.лог("Доступ разрешен");
} иначе если (возраст >= 16) {
  консоль.лог("Доступ с ограничениями");
} иначе {
  консоль.лог("Доступ запрещен");
}
```

## Циклы

// Цикл for

## Функции

```javascript
// Объявление функции
функция привет(имя = 'Гость') {
  вернуть `Привет, ${имя}!`;
}

// Стрелочная функция
конст сложить = (а, б) => а + б;

// Асинхронная функция
асинк функция получитьДанные() {
  ждать задержка(1000);
  вернуть { данные: 'готово' };
}
```

## Классы

```javascript
класс Животное {
  конструктор(имя) {
    this.имя = имя;
  }

  говори() {
    консоль.лог(`${this.имя} издает звук.`);
  }
}

класс Собака расширяет Животное {
  лает() {
    консоль.лог('Гав-гав!');
  }
}

const мойПес = новый Собака('Рекс');
мойПес.говори(); // Рекс издает звук.
мойПес.лает();    // Гав-гав!
```
